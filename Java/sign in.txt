    @SuppressWarnings("SuspiciousIndentAfterControlStatement")
    private void jButton1MouseClicked(java.awt.event.MouseEvent evt) {                                      
        String Name = jTextField1.getText();
        String Email = jTextField2.getText();
        char ch[] = jPasswordField1.getPassword();
        String str = new String(ch);
        String Gender = "";
        if(jRadioButton1.isSelected())
        Gender = "Male";
        if(jRadioButton2.isSelected())
        Gender = "Female";
        String SelectArea = (String)jComboBox1.getSelectedItem();
        String MobileNumber = jTextField3.getText();
        String OTP = jTextField4.getText();
        if(Name.equals(""))
        {
            jLabel9.setText("name field is mandatory");
            
        }
        else if(str.equals(""))
        {
            jLabel10.setText("password field is mandatory");
        }
        else{
            
        
        try
        {
            Class.forName("com.mysql.jdbc.Driver");
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/test","root","root");
            PreparedStatement st = con.prepareStatement("insert into signup values(?,?,?,?,?,?,?)");
            st.setString(1,Name);
            st.setString(2,Email);
            st.setString(3,str);
            st.setString(4,Gender);
            st.setString(5,SelectArea);
            st.setString(6,MobileNumber);
            st.setString(7,OTP);
            st.executeUpdate();
            JOptionPane.showMessageDialog(this,"Done");
            jTextField1.setText("");
            jTextField2.setText("");
            jPasswordField1.setText("");
            buttonGroup1.clearSelection();
            jTextField3.setText("");
            jTextField4.setText("");
       
        }
        catch(SQLException | ClassNotFoundException ex)
        {
            ex.printStackTrace();
        }
    }                                     
    }
