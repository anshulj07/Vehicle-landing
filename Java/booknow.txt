    private void jButton2MouseClicked(java.awt.event.MouseEvent evt) {                                      
          String Bank = (String)jComboBox1.getSelectedItem();
          String Account = jTextField1.getText();
          String ifsc = jTextField2.getText();
          String holder = jTextField3.getText();
          String mobile = jTextField4.getText();
          String captcha = jTextField5.getText();
          if(Account.equals(""))
          {
            jLabel10.setText("Enter Account Number");
          }
          if(ifsc.equals(""))
          {
            jLabel11.setText("Enter Ifsc Code");
          }
          if(holder.equals(""))
          {
            jLabel12.setText("Enter Holder Name");
          }
          if(mobile.equals(""))
          {
            jLabel13.setText("Enter Mobile Number");
          }else 
          {
             try
          {
             Class.forName("com.mysql.jdbc.Driver");
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/test","root","root");
            PreparedStatement st = con.prepareStatement("insert into payment values(?,?,?,?,?,?)");
            st.setString(1,Bank);
            st.setString(2,Account);
            st.setString(3,ifsc);
            st.setString(4,holder);
            st.setString(5,mobile);
            st.setString(6,captcha);
            st.executeUpdate();
            JOptionPane.showMessageDialog(this,"Done");
            jTextField1.setText("");
            jTextField2.setText("");
            jTextField3.setText("");
            jTextField4.setText(""); 
            jTextField5.setText("");
            
            int Result = 0;
            if(Result == 0)
            {
              new Confirm().setVisible(true);
              dispose();
            }
          }
          catch(SQLException | ClassNotFoundException ex)
          {
             ex.printStackTrace();
          }
          
          }
    }
